// router.c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>
#include <pthread.h>

#define BUFFER_SIZE 256
#define PORT 4500
#define SERVIDOR_IP "127.0.0.1"
#define SERVIDOR_PORT 5500

void *handle_client(void *socket_cliente_ptr) {
    int socket_cliente = *(int *)socket_cliente_ptr;
    free(socket_cliente_ptr);

    char buffer[BUFFER_SIZE];
    ssize_t num_bytes;

    // Recibir mensaje del cliente
    num_bytes = read(socket_cliente, buffer, BUFFER_SIZE);
    if (num_bytes < 0) {
        perror("Error al leer del socket cliente");
        close(socket_cliente);
        pthread_exit(NULL);
    }

    // Conectar al servidor principal para enviar el mensaje
    int socket_servidor = socket(AF_INET, SOCK_STREAM, 0);
    if (socket_servidor < 0) {
        perror("Error al crear el socket del servidor");
        close(socket_cliente);
        pthread_exit(NULL);
    }

    struct sockaddr_in dir_servidor;
    memset(&dir_servidor, 0, sizeof(dir_servidor));
    dir_servidor.sin_family = AF_INET;
    dir_servidor.sin_addr.s_addr = inet_addr(SERVIDOR_IP);
    dir_servidor.sin_port = htons(SERVIDOR_PORT);

    if (connect(socket_servidor, (struct sockaddr *)&dir_servidor, sizeof(dir_servidor)) < 0) {
        perror("Error al conectar con el servidor principal");
        close(socket_cliente);
        close(socket_servidor);
        pthread_exit(NULL);
    }

    // Enviar mensaje al servidor principal
    write(socket_servidor, buffer, num_bytes);

    // Recibir respuesta del servidor principal
    num_bytes = read(socket_servidor, buffer, BUFFER_SIZE);
    if (num_bytes < 0) {
        perror("Error al leer del socket servidor");
        close(socket_cliente);
        close(socket_servidor);
        pthread_exit(NULL);
    }

    // Enviar respuesta al cliente
    write(socket_cliente, buffer, num_bytes);

    close(socket_cliente);
    close(socket_servidor);
    pthread_exit(NULL);
}

int main() {
    int socket_router, *socket_cliente_ptr;
    struct sockaddr_in dir_router, dir_cliente;
    socklen_t tam_dir_cliente = sizeof(dir_cliente);

    socket_router = socket(AF_INET, SOCK_STREAM, 0);
    if (socket_router < 0) {
        perror("Error al crear el socket del router");
        exit(EXIT_FAILURE);
    }

    memset(&dir_router, 0, sizeof(dir_router));
    dir_router.sin_family = AF_INET;
    dir_router.sin_addr.s_addr = htonl(INADDR_ANY);
    dir_router.sin_port = htons(PORT);

    if (bind(socket_router, (struct sockaddr *)&dir_router, sizeof(dir_router)) < 0) {
        perror("Error al vincular el socket del router");
        close(socket_router);
        exit(EXIT_FAILURE);
    }

    listen(socket_router, 5);

    while (1) {
        socket_cliente_ptr = malloc(sizeof(int));
        *socket_cliente_ptr = accept(socket_router, (struct sockaddr *)&dir_cliente, &tam_dir_cliente);
        if (*socket_cliente_ptr < 0) {
            perror("Error al aceptar la conexiÃ³n del cliente");
            free(socket_cliente_ptr);
            continue;
        }

        pthread_t thread_id;
        if (pthread_create(&thread_id, NULL, handle_client, socket_cliente_ptr) != 0) {
            perror("Error al crear el hilo para el cliente");
            close(*socket_cliente_ptr);
            free(socket_cliente_ptr);
        }
    }

    close(socket_router);
    return 0;
}
